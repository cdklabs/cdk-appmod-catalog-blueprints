1. Create `src/context/ChatContext.tsx`:

```typescript
import { createContext, useContext, useReducer, useCallback, ReactNode } from 'react';
import { ChatState, ChatAction, Message, SchemaColumn, DownloadLinks } from '../types';
import { sendMessage as apiSendMessage, SSECallbacks } from '../services/api';
import { v4 as uuidv4 } from 'uuid'; // Add uuid to package.json dependencies

const initialState: ChatState = {
  messages: [],
  schema: [],
  preview: [],
  downloads: null,
  isStreaming: false,
  error: null,
};

function chatReducer(state: ChatState, action: ChatAction): ChatState {
  switch (action.type) {
    case 'ADD_USER_MESSAGE':
      return { ...state, messages: [...state.messages, action.payload] };
    case 'START_STREAMING':
      // Add empty assistant message that will be appended to
      return {
        ...state,
        isStreaming: true,
        messages: [...state.messages, { id: 'streaming', role: 'assistant', content: '', timestamp: new Date() }],
      };
    case 'APPEND_CONTENT':
      // Append to last message (streaming assistant message)
      const updatedMessages = [...state.messages];
      const lastIndex = updatedMessages.length - 1;
      updatedMessages[lastIndex] = {
        ...updatedMessages[lastIndex],
        content: updatedMessages[lastIndex].content + action.payload,
      };
      return { ...state, messages: updatedMessages };
    case 'SET_SCHEMA':
      return { ...state, schema: action.payload };
    case 'SET_PREVIEW':
      return { ...state, preview: action.payload };
    case 'SET_DOWNLOADS':
      return { ...state, downloads: action.payload };
    case 'STOP_STREAMING':
      // Finalize the streaming message with a real ID
      const finalMessages = [...state.messages];
      const lastMsg = finalMessages[finalMessages.length - 1];
      if (lastMsg?.id === 'streaming') {
        finalMessages[finalMessages.length - 1] = { ...lastMsg, id: uuidv4() };
      }
      return { ...state, isStreaming: false, messages: finalMessages };
    case 'SET_ERROR':
      return { ...state, error: action.payload, isStreaming: false };
    case 'CLEAR_ERROR':
      return { ...state, error: null };
    default:
      return state;
  }
}

interface ChatContextValue {
  state: ChatState;
  sendMessage: (content: string) => Promise<void>;
}

const ChatContext = createContext<ChatContextValue | null>(null);

export function ChatProvider({ children }: { children: ReactNode }) {
  const [state, dispatch] = useReducer(chatReducer, initialState);
  const sessionId = useRef(uuidv4()).current; // Session ID persists for conversation

  const sendMessage = useCallback(async (content: string) => {
    // Add user message
    const userMessage: Message = {
      id: uuidv4(),
      role: 'user',
      content,
      timestamp: new Date(),
    };
    dispatch({ type: 'ADD_USER_MESSAGE', payload: userMessage });
    dispatch({ type: 'START_STREAMING' });
    dispatch({ type: 'CLEAR_ERROR' });

    const callbacks: SSECallbacks = {
      onMessage: (chunk) => dispatch({ type: 'APPEND_CONTENT', payload: chunk }),
      onSchema: (schema) => dispatch({ type: 'SET_SCHEMA', payload: schema }),
      onPreview: (data) => dispatch({ type: 'SET_PREVIEW', payload: data }),
      onDownload: (links) => dispatch({ type: 'SET_DOWNLOADS', payload: links }),
      onError: (error) => dispatch({ type: 'SET_ERROR', payload: error }),
      onDone: () => dispatch({ type: 'STOP_STREAMING' }),
    };

    try {
      await apiSendMessage(content, sessionId, callbacks);
    } catch (error) {
      dispatch({ type: 'SET_ERROR', payload: error instanceof Error ? error.message : 'Unknown error' });
    }
  }, [sessionId]);

  return (
    <ChatContext.Provider value={{ state, sendMessage }}>
      {children}
    </ChatContext.Provider>
  );
}

export function useChatContext() {
  const context = useContext(ChatContext);
  if (!context) {
    throw new Error('useChatContext must be used within ChatProvider');
  }
  return context;
}
```
